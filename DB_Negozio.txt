create database negoziorossi;

use negoziorossi;

create table utenti(
id int primary key auto_increment,
nome varchar(50),
cognome varchar(50),
cellulare varchar(50),
indirizzo varchar(50)
);

create table prodotti(
id int primary key auto_increment,
nome varchar(50),
categoria varchar(50),
prezzo double
);

create table stock(
id int primary key auto_increment,
colore varchar(50),
taglia varchar(50),
idprodotto int,
foreign key (idprodotto) references prodotti(id)
);

create table credenziali(
id int primary key auto_increment,
mail varchar(50),
psw varchar(50),
idutente int unique,
foreign key (idutente) references utenti(id)
);

create table recensione(
id int primary key auto_increment,
idprodotto int,
foreign key (idprodotto) references prodotti(id),
idutente int,
foreign key (idutente) references utenti(id),
commento varchar(200),
valutazione int
);

create table prodotti_in_ordine(
id int primary key auto_increment,
idprodotto int,
foreign key (idprodotto) references prodotti(id),
npezzi int
);

create table ordini(
id int primary key auto_increment,
idcarrello int,
foreign key (idprodotto) references prodotti_in_ordine(id),
idutente int,
foreign key (idutente) references utenti(id),
prezzo double
);
